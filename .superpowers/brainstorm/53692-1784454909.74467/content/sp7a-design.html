2. SkillLoader 组件

新增 agentteam/runtime/skills.py
"""Skill 加载器:从 skills/ 目录读取 .md 文件,提供按名查询接口。"""
from pathlib import Path


class SkillLoader:
    """从文件系统加载 markdown skill 文件,内存缓存避免重读。

    skill 名 = 文件名 stem(如 skills/code_review.md → "code_review")。
    若 skill 不存在,load 时抛 KeyError(编译期 fail-fast,避免运行时静默缺失)。
    """

    def __init__(self, skills_dir: Path | None = None) -> None:
        self._skills_dir = skills_dir
        self._cache: dict[str, str] = {}    # name → md_content
        self._scanned = False

    def _scan(self) -> None:
        """惰性扫描 skills_dir,构建 name → content 缓存。"""
        if self._scanned or self._skills_dir is None:
            self._scanned = True
            return
        for path in self._skills_dir.glob("*.md"):
            self._cache[path.stem] = path.read_text(encoding="utf-8")
        self._scanned = True

    def load(self, names: list[str]) -> dict[str, str]:
        """按名批量加载 skill,返回 {name: content}。

        缺失的 skill 名抛 KeyError(列出缺失项便于诊断)。
        空名列表返回空 dict(默认场景,Agent.skills == [])。
        """
        if not names:
            return {}
        self._scan()
        missing = [n for n in names if n not in self._cache]
        if missing:
            raise KeyError(f"Skills not found: {missing}")
        return {n: self._cache[n] for n in names}

    def list_available(self) -> list[str]:
        """列出所有可用 skill 名(供 API / CLI 查询)。"""
        self._scan()
        return sorted(self._cache.keys())

    def reload(self) -> None:
        """清缓存重扫(支持 skills/ 目录热更新)。"""
        self._cache.clear()
        self._scanned = False
        self._scan()

    def _compile_worker(self, agent: Agent, default_model, trace_writer, audit_repo):
        # 加载该 agent 装备的 skills
        skills = self._skill_loader.load(agent.skills)
        llm = self._mp.get_llm(agent.model or default_model)
        tools = self._tr.get_tools(agent.tools) if agent.tools else []

            agent, llm, tools, trace_writer, audit_repo,
            run_manager=self._run_manager,
            skills=skills,  # 透传给 worker subgraph
        )

4. Skill 注入流程(关键)

runtime/nodes.py — make_init_worker 把 skills 注入 react_messages
def make_init_worker(agent: Agent, trace_writer=None, skills: dict[str, str] | None = None):
    def init_worker(state: dict) -> dict:
        # 原有初始化逻辑(system_prompt + task)
        react_messages = [
            SystemMessage(content=agent.system_prompt),
            HumanMessage(content=state.get("task", "")),
        ]

        # SP7a: 把 skills 包装为 SystemMessage 插入 react_messages 首部之后
        if skills:
            skill_text = "\n\n".join(
                f'\n{content}\n'
                for name, content in skills.items()
            )
            react_messages.insert(1, SystemMessage(content=skill_text))
            # ↑ insert(1, ...) 让 skill 在 system_prompt 之后、task 之前

        return {
            "react_messages": react_messages,
            "tool_calls": [],
            "iteration": 0,
            "final_answer": "",
            # dag 模式字段...
        }
    return init_worker

5. Skill 注入位置示意图

react_messages 结构对比
【改造前】react_messages 结构:
┌──────────────────────────────────┐
│ SystemMessage(system_prompt)     │  ← Agent 身份描述
├──────────────────────────────────┤
│ HumanMessage(task)               │  ← 用户任务
├──────────────────────────────────┤
│ AIMessage / ToolMessage ...      │  ← ReAct 循环
└──────────────────────────────────┘

【改造后】react_messages 结构(有 skills 时):
┌──────────────────────────────────┐
│ SystemMessage(system_prompt)     │  ← Agent 身份描述(不变)
├──────────────────────────────────┤
│ SystemMessage(skill_text)         │  ← SP7a 新增:行为指导
│   <skill name="code_review">     │
│   审查代码时先检查...             │
│   </skill>                       │
│   <skill name="testing">         │
│   测试覆盖率达到...               │
│   </skill>                       │
├──────────────────────────────────┤
│ HumanMessage(task)               │  ← 用户任务
├──────────────────────────────────┤
│ AIMessage / ToolMessage ...      │  ← ReAct 循环
└──────────────────────────────────┘

设计理由:
- skill 放在 system_prompt 之后:LLM 先建立身份,再接收行为指导
- skill 放在 task 之前:LLM 处理任务时已"知道"该用什么模式
- 用 <skill> 标签包裹:便于 LLM 区分多个 skill 边界
- 不混入 system_prompt:身份描述与行为指导解耦,便于独立演进
